Create Request Models

🤔 Why separate request structs? We don't want users sending "id": 999 or "created_at": "random" during creation!

Think of it like a job application form 📋 — you only fill in YOUR info, not the company's internal ID or hire date!

package models

import "time"

// Database Model — represents a ROW in PostgreSQL
type Superhero struct {
    ID         int       `json:"id"`
    Name       string    `json:"name"`
    RealName   string    `json:"real_name"`
    Universe   string    `json:"universe"`
    PowerLevel int       `json:"power_level"`
    CreatedAt  time.Time  `json:"created_at"`
}

// Request Model — what the CLIENT sends
type CreateSuperheroRequest struct {
    Name       string `json:"name"`
    RealName   string `json:"real_name"`
    Universe   string `json:"universe"`
    PowerLevel int    `json:"power_level"`
}

// Request Model — for updates (same fields, different purpose!)
type UpdateSuperheroRequest struct {
    Name       string `json:"name"`
    RealName   string `json:"real_name"`
    Universe   string `json:"universe"`
    PowerLevel int    `json:"power_level"`
}

🗄️ Superhero

Represents a Database Row — includes auto-generated id and created_at

📨 Create/Update Request

Represents Incoming JSON — only fields the user should provide

Complete the Repository

💡 New Methods: Create(), GetAll(), GetByID(), Update(), Delete() — the full CRUD set!

package repository

import (
    "database/sql"
    "superhero-api/internal/models"
)

type SuperheroRepository struct {
    DB *sql.DB
}

func NewSuperheroRepository(db *sql.DB) *SuperheroRepository {
    return &SuperheroRepository{DB: db}
}

// CREATE — Exec() because no rows are returned
func (r *SuperheroRepository) Create(req models.CreateSuperheroRequest) error {
    _, err := r.DB.Exec(`
        INSERT INTO superheroes
        (name, real_name, universe, power_level)
        VALUES ($1,$2,$3,$4)
    `, req.Name, req.RealName, req.Universe, req.PowerLevel)
    return err
}

// READ ALL — Query() because multiple rows
func (r *SuperheroRepository) GetAll() ([]models.Superhero, error) {
    rows, err := r.DB.Query(`
        SELECT id, name, real_name, universe, power_level, created_at
        FROM superheroes ORDER BY id
    `)
    if err != nil { return nil, err }
    defer rows.Close()

    var heroes []models.Superhero
    for rows.Next() {
        var hero models.Superhero
        err := rows.Scan(&hero.ID, &hero.Name, &hero.RealName,
            &hero.Universe, &hero.PowerLevel, &hero.CreatedAt)
        if err != nil { return nil, err }
        heroes = append(heroes, hero)
    }
    return heroes, nil
}

// READ ONE — QueryRow() because exactly one row
func (r *SuperheroRepository) GetByID(id int) (models.Superhero, error) {
    var hero models.Superhero
    err := r.DB.QueryRow(`
        SELECT id, name, real_name, universe, power_level, created_at
        FROM superheroes WHERE id=$1
    `, id).Scan(&hero.ID, &hero.Name, &hero.RealName,
        &hero.Universe, &hero.PowerLevel, &hero.CreatedAt)
    return hero, err
}

// UPDATE — Exec() because no rows returned
func (r *SuperheroRepository) Update(id int, req models.UpdateSuperheroRequest) error {
    _, err := r.DB.Exec(`
        UPDATE superheroes
        SET name=$1, real_name=$2, universe=$3, power_level=$4
        WHERE id=$5
    `, req.Name, req.RealName, req.Universe, req.PowerLevel, id)
    return err
}

// DELETE — Exec() because no rows returned
func (r *SuperheroRepository) Delete(id int) error {
    _, err := r.DB.Exec(`
        DELETE FROM superheroes WHERE id=$1
    `, id)
    return err
}

🔍 Deep Dive: Exec() vs QueryRow() vs Query()

Method Returns Use When SQL Examples
Exec() sql.Result No rows returned INSERT, UPDATE, DELETE
QueryRow() *sql.Row Exactly one row SELECT ... WHERE id=1
Query() *sql.Rows Multiple rows SELECT * FROM ...

Create the Handler Layer

🎭 Handlers are like waiters at a restaurant — they take your order (HTTP request), pass it to the kitchen (repository), and bring back your food (JSON response)!

package handlers

import (
    "net/http"
    "strconv"
    "github.com/gin-gonic/gin"
    "superhero-api/internal/models"
    "superhero-api/internal/repository"
)

type SuperheroHandler struct {
    Repo *repository.SuperheroRepository
}

func NewSuperheroHandler(repo *repository.SuperheroRepository) *SuperheroHandler {
    return &SuperheroHandler{Repo: repo}
}

// POST /superheroes — Create a new hero!
func (h *SuperheroHandler) Create(c *gin.Context) {
    var req models.CreateSuperheroRequest

    // 🧲 ShouldBindJSON = magic decoder!
    // Takes raw JSON body → fills the struct automatically
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }

    if err := h.Repo.Create(req); err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
        return
    }

    c.JSON(http.StatusCreated, gin.H{"message": "superhero created"})
}

// GET /superheroes — Get ALL heroes!
func (h *SuperheroHandler) GetAll(c *gin.Context) {
    heroes, err := h.Repo.GetAll()
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
        return
    }
    c.JSON(http.StatusOK, heroes)
}

// GET /superheroes/:id — Get ONE hero by ID!
func (h *SuperheroHandler) GetByID(c *gin.Context) {
    idParam := c.Param("id")
    id, err := strconv.Atoi(idParam)
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": "invalid id"})
        return
    }

    hero, err := h.Repo.GetByID(id)
    if err != nil {
        c.JSON(http.StatusNotFound, gin.H{"error": "hero not found"})
        return
    }

    c.JSON(http.StatusOK, hero)
}

// PUT /superheroes/:id — Update a hero!
func (h *SuperheroHandler) Update(c *gin.Context) {
    id, err := strconv.Atoi(c.Param("id"))
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": "invalid id"})
        return
    }

    var req models.UpdateSuperheroRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }

    if err := h.Repo.Update(id, req); err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
        return
    }

    c.JSON(http.StatusOK, gin.H{"message": "updated"})
}

// DELETE /superheroes/:id — Remove a hero!
func (h *SuperheroHandler) Delete(c *gin.Context) {
    id, err := strconv.Atoi(c.Param("id"))
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": "invalid id"})
        return
    }

    if err := h.Repo.Delete(id); err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
        return
    }

    c.JSON(http.StatusOK, gin.H{"message": "deleted"})
}

✅ What just happened? Every handler follows the same pattern: Parse input → Call Repo → Return JSON. Like a recipe you can repeat!

🧲 ShouldBindJSON() Explained

It's like a magnet that pulls JSON from the request body and sticks it into your Go struct! Behind the scenes, it's json.Unmarshal() with extra Gin magic ✨

Input JSON:

{"name":"Thor","real_name":"Thor Odinson","universe":"Marvel","power_level":99}

Output: A fully populated CreateSuperheroRequest struct!

Create Routes

🛣️ Why a separate routes file? Without it, main.go would be 1000 lines of messy route definitions. Routes stay organized and clean!

package routes

import (
    "github.com/gin-gonic/gin"
    "superhero-api/internal/handlers"
)

func RegisterRoutes(router *gin.Engine, handler *handlers.SuperheroHandler) {
    router.POST("/superheroes", handler.Create)
    router.GET("/superheroes", handler.GetAll)
    router.GET("/superheroes/:id", handler.GetByID)
    router.PUT("/superheroes/:id", handler.Update)
    router.DELETE("/superheroes/:id", handler.Delete)
}

🗺️ URL Pattern Cheat Sheet

/superheroes → All heroes
/superheroes/:id → One hero (replace :id with number)

Wire It All Together in main.go

🔌 The Grand Assembly! Connect config → database → repository → handler → router → run server!

package main

import (
    "log"
    "github.com/gin-gonic/gin"
    "superhero-api/internal/config"
    "superhero-api/internal/database"
    "superhero-api/internal/handlers"
    "superhero-api/internal/repository"
    "superhero-api/internal/routes"
)

func main() {
    // 1. Load config (like reading a settings file)
    cfg := config.Load()

    // 2. Connect to PostgreSQL (open the database door)
    db, err := database.NewPostgresDB(cfg)
    if err != nil {
        log.Fatal(err)
    }

    // 3. Create repository (the kitchen)
    repo := repository.NewSuperheroRepository(db)

    // 4. Create handler (the waiter)
    handler := handlers.NewSuperheroHandler(repo)

    // 5. Create router (the menu)
    router := gin.Default()

    // 6. Register routes (put items on the menu)
    routes.RegisterRoutes(router, handler)

    // 7. Start the server! 🚀
    log.Println("Server running on :8080")
    router.Run(":8080")
}

🎉 Success! Your server is now listening on port 8080, ready to accept HTTP requests!
